### Customized colors

Though probably not the final official way to do it, this is a way to achieve one-off alternative
styles for buttons.

```jsx
<Button
    variant="primary"
    style={{
        '--reactist-btn-idle-tint': '#eee',
        '--reactist-btn-idle-fill': '#666',
        '--reactist-btn-hover-tint': '#fff',
        '--reactist-btn-hover-fill': '#777',
        '--reactist-btn-disabled-tint': '#fff',
        '--reactist-btn-disabled-fill': '#ccc',
    }}
>
    Click me
</Button>
```

export function Template({ label, variant, tone, size }) {
    if (label === 'Click me <strong>now</strong>') {
        label = (
            <span>
                Click me <strong key="strong">now</strong>
            </span>
        )
    }
    return (
        <Inline space="large">
            <Stack space="large">
                <Box>
                    <Button variant={variant} tone={tone} size={size}>
                        {label}
                    </Button>
                </Box>
                <Box>
                    <Button variant={variant} tone={tone} size={size} disabled>
                        {label}
                    </Button>
                </Box>
            </Stack>
            <Stack space="large">
                <Box>
                    <Button variant={variant} tone={tone} size={size}>
                        <Icon />
                        {label}
                    </Button>
                </Box>
                <Box>
                    <Button variant={variant} tone={tone} size={size} disabled>
                        <Icon />
                        {label}
                    </Button>
                </Box>
            </Stack>
            <Stack space="large">
                <Box>
                    <Button variant={variant} tone={tone} size={size}>
                        {label}
                        <Icon />
                    </Button>
                </Box>
                <Box>
                    <Button variant={variant} tone={tone} size={size} disabled>
                        {label}
                        <Icon />
                    </Button>
                </Box>
            </Stack>
            <Stack space="large">
                <Box>
                    <Button variant={variant} tone={tone} size={size} tooltip={label}>
                        <Icon />
                    </Button>
                </Box>
                <Box>
                    <Button variant={variant} tone={tone} size={size} disabled tooltip={label}>
                        <Icon />
                    </Button>
                </Box>
            </Stack>
        </Inline>
    )
}
